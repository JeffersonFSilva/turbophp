  278.   /**
    279. 

  279.    * Perform data acess.
    280. 

  280.    * Fires {@link OnBeforeExecute} then calls {@link ExecuteQuery}.
    281. 

  281.    * Subsequently fires {@link OnFailure} or {@link OnSuccess} depending
    282. 

  282.    * on the value of {@link Result}.
    283. 

  283.    * This function is called automatically if data access
    284. 

  284.    * functions are invoked and data is not yet available.
    285. 

  285.    * Specifically {@link Next}, {@link Rewind} and {@link GetField}
    286. 

  286.    * will call Execute if needed.
    287. 

  287.    */
    288. 

  288.   function Execute()
    289. 

  289.   {
    290. 

  290.     $this->DoEvent($this->OnBeforeExecute);
    291. 

  291.     $this->DoExecute();
    292. 

  292.     if (!$this->Result)
    293. 

  293.       $this->DoEvent($this->OnFailure);
    294. 

  294.     else
    295. 

  295.       $this->DoEvent($this->OnSuccess);
    296. 

  296.   }
    297. 

  297.   /**
    298. 

  298.    * @access private
    299. 

  299.    */
    300. 

  300.   function NeedData()
    301. 

  301.   {
    302. 

  302.     if (!$this->Result)
    303. 

  303.       $this->Execute();
    304. 

  304.   }
    305. 

  305.   /**
    306. 

  306.    * True if the end of data has been reached.
    307. 

  307.    * @return boolean 
    308. 

  308.    */
    309. 

  309.   function EOF()
    310. 

  310.   {
    311. 

  311.     return !$this->Row;
    312. 

  312.   }
    313. 

  313.   /**

    316. 

  316.    */
    317. 

  317.   function Rewind()
    318. 

  318.   {
    319. 

  319.     $this->NeedData();
    320. 

  320.   }
    321. 

  321.   /**

  322.    * @abtract
    323. 

  323.    * @access private
    324. 

  324.    */
    325. 

  325.   function DoNextRow()
    326. 

  326.   {
    327. 

  327.   }
    328. 

  328.   /**
    329. 

  329.    * Fill {@link $Row} with the next row of data.

  330.    * Calls {@link Execute} if needed.
    331. 

  331.    */
    332. 

  332.   function Next()
    333. 

  333.   {
    334. 

  334.     $this->NeedData();
    335. 

  335.     if ($this->Result)
    336. 

  336.       $this->DoNextRow();
    337. 

  337.     return !$this->EOF();
    338. 

  338.   }
    339. 

  339.   /**
    340. 

  340.    * Get field data from the current row.
    341. 

  341.    * Calls {@link Execute} if needed.
    342. 

  342.    * @return Value of key $inFieldId in the array {@link $Row}, or false

  343.    *  if the key does not exist.

  344.    */
    345.

    346. 

  346.   {
    347. 

  347.     $this->NeedData();
    348. 

  348.     if (!$this->Row)
    349. 

  349.       return '(bad row)';
    350. 

  350.     else
    351. 

  351.       return TpSafeArrayGet($inFieldId, $this->Row);
    352. 

  352.   }
    353. 

  353.   /**
    354. 

  354.    * Add an entire column of the result data to $outItems.

  355.    * The column identified by the {@link $Field} property is copied.

  356.    * By default, the first column of the data set is used.

    364.

    365. 

  365.       $outItems[] = @$this->Row[$this->Field];
    366. 

  366.   }
    367. 

  367. }

  372. class TTpQuery extends TTpDataSource
    373. 

  373. {
    374. 

  374.   /**
    375. 

  375.    * SQL to be executed.

  376.    * @var string 
    377. 

  377.    */
    378. 

  378.   var $SQL;
    379. 

  379.   /**
    380. 

  380.    * Name, value pairs of SQL parameters.
    381. 

  381.    * The {@link $SQL} for this control may contain parameters (macros) of the
    382. 

  382.    * form {%<param_name>}. These parameters are replaced by values from the
    383. 

  383.    * $Params array.
    384. 

  384.    *
    385. 

  385.    * Example 1: the SQL contains a line like
    386. 

  386.    *  where Cost > {%MinCost}
    387. 

  387.    * and the $Params array has
    388. 

  388.    *  "MinCost" => "1000"
    389. 

  389.    * the executed SQL will read
    390. 

  390.    *  where Cost > 1000
    391. 

  391.    *
    392. 

  392.    * Example 2: the SQL contains a line like
    393. 

  393.    *  order by '{%SortField}'
    394. 

  394.    * and the $Params array has
    395. 

  395.    *  "SortField" => "Name"
    396. 

  396.    * the executed SQL will read
    397. 

  397.    *  order by 'Name'
    398. 

  398.    *
    399. 

  399.    * @var array 
    400. 

  400.    */
    401. 

  401.   var $Params;
    402. 

  402.   /**
    403. 

  403.    * @access private
    404. 

  404.    */
    405. 

  405.   function Init()
    406. 

  406.   {
    407. 

  407.     parent::Init();
    408. 

  408.     $this->Params = array();
    409. 

  409.   }
    410. 

  410.   /**
    411. 

  411.    * @access private
    412. 

  412.    */
    413. 

  413.   function AfterParse()
    414. 

  414.   {
    415. 

  415.     parent::AfterParse();
    416. 

  416.     $this->SQL = base64_decode($this->SQL);
    417. 

  417.   }
    418. 

  418.   /**
    419. 

  419.    * @access private
    420. 

  420.    */
    421. 

  421.   function InterpolateParam(&$matches)
    422. 

  422.   {
    423. 

  423.     $obj = $this->App->GetObject($matches[1]);
    424. 

  424.     if ($obj)
    425. 

  425.       $p = $obj->GetDefaultValue();
    426. 

  426.     else
    427. 

  427.       $p = TpSafeArrayGet($matches[1], $this->Params);
    428. 

  428.     return $this->Interpolate($p);
    429. 

  429.   }
    430. 

  430.   /**
    431. 

  431.    * @access private
    432. 

  432.    */
    433. 

  433.   function Interpolate(&$inText)
    434. 

  434.   {
    435. 

  435.     return preg_replace_callback(
    436. 

  436.       P_QUERY_PARAM,
    437. 

  437.       array(&$this, 'InterpolateParam'),
    438. 

  438.       $inText
    439. 

  439.     );
    440. 

  440.   }
    441. 

  441.   /**
    442. 

  442.    * @access private
    443. 

  443.    */
    444. 

  444.   function InterpolatedQuery()
    445. 

  445.   {
    446. 

  446.     return $this->Interpolate($this->SQL);
    447. 

  447.   }
    448. 

  448.   function Dump()
    449. 

  449.   {
    450. 

  450.     parent::Dump();
    451. 

  451.     echo "SQL: [$this->SQL]<br>";
    452. 

  452.     if (count($this->Params) > 0)
    453. 

  453.     {
    454. 

  454.       print_r($this->Params);
    455. 

  455.       echo "<br>";
    456. 

  456.     }
    457. 

  457.   }
    458.
