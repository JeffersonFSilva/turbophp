  34. /**#@-*/ * Create an SQL insert command.
    35. 

  35.  * Converts a table name and an array of name-value pairs

  36.  * into an SQL insert statment.

  37.  * @param $inTable string Table name for the query.
    38. 

  38.  * @param $inData array Data to insert, array of (fieldname => value) records.
    39. 

  39.  * @return string SQL insert command.

  40.  */
    41. 

  41. function tpGetInsertSql($inTable, $inData)
    42. 

  42. {
    43. 

  43.   $p = implode(",", array_keys($inData));
    44. 

  44.   $v = "'" . implode("','", array_values($inData)) . "'";
    45. 

  45.   return sprintf("INSERT INTO %s (%s) VALUES (%s)", $inTable, $p, $v);
    46. 

  46. }
